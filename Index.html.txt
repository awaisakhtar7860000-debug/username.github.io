window.addEventListener('load', function(){
  fetchIp(function(){
    const savedToken = localStorage.getItem(SESSION_KEY);
    if(savedToken){
      state.processing = true; render();
      google.script.run.withSuccessHandler(function(res){
        state.processing = false;
        if(res.valid){
          state.user = res.user; state.token = savedToken;
          state.activeRole = localStorage.getItem(activeRoleKey_()) || (res.user.roles && res.user.roles[0]) || res.user.role;
          ensureSessionExpiry_();
          if(state.user.email){ enterDashboard(); }
          else { state.view = 'email_gate'; render(); logPage('email_gate'); }
        } else {
          localStorage.removeItem(SESSION_KEY);
          localStorage.removeItem(SESSION_EXPIRY_KEY);
          state.view='login'; render();
        }
      }).withFailureHandler(function(){ state.processing=false; state.view='login'; render(); })
        .checkSession(savedToken, detectDevice(), clientIp);
    } else {
      state.view='login'; render();
    }
  });
  startSessionWatcher();
});

let sessionWatchHandle = null;
function startSessionWatcher(){
  clearInterval(sessionWatchHandle);
  sessionWatchHandle = setInterval(function(){
    if(!state.user || !state.token) return;
    const exp = Number(localStorage.getItem(SESSION_EXPIRY_KEY) || 0);
    if(exp && Date.now() > exp){
      stopAutoSync();
      google.script.run.logoutSession(state.token);
      localStorage.removeItem(SESSION_KEY);
      localStorage.removeItem(SESSION_EXPIRY_KEY);
      localStorage.removeItem(LAST_VIEW_KEY);
      state = { view:'login', user:null, token:null, processing:false, selectedCandidate:null, voteConfirm:null, notifications:[], notifHistory:null, settingsSaving:{}, reportBusy:{} };
      render();
      showToast('Your session has expired after ' + SESSION_MAX_HOURS + ' hours. Please log in again.', 'info');
    }
  }, SESSION_CHECK_INTERVAL_MS);
}

function collectOtpValue_(){
  let val = '';
  for(let i=0;i<OTP_LENGTH;i++){
    const el = document.getElementById('otpBox'+i);
    val += el ? (el.value||'').trim() : '';
  }
  return val;
}
function focusOtpBox_(i){
  const el = document.getElementById('otpBox'+i);
  if(el) el.focus();
}
function markOtpBoxesError_(hasError){
  for(let i=0;i<OTP_LENGTH;i++){
    const el = document.getElementById('otpBox'+i);
    if(el) el.classList.toggle('error', !!hasError);
  }
}
function onOtpBoxInput(i, el){
  el.value = el.value.replace(/[^0-9]/g,'').slice(0,1);
  markOtpBoxesError_(false);
  if(el.value && i < OTP_LENGTH-1){ focusOtpBox_(i+1); }
  if(el.value && i === OTP_LENGTH-1){
    const full = collectOtpValue_();
    if(full.length === OTP_LENGTH) onVerifyOtp();
  }
}
function onOtpBoxKeydown(i, ev){
  if(ev.key === 'Backspace' && !ev.target.value && i > 0){ focusOtpBox_(i-1); }
}
function onOtpPaste(ev){
  const text = (ev.clipboardData || window.clipboardData).getData('text').replace(/[^0-9]/g,'').slice(0,OTP_LENGTH);
  if(!text) return;
  ev.preventDefault();
  for(let i=0;i<OTP_LENGTH;i++){
    const el = document.getElementById('otpBox'+i);
    if(el) el.value = text[i] || '';
  }
  if(text.length === OTP_LENGTH) onVerifyOtp();
  else focusOtpBox_(text.length);
}

function onVerifyOtp(){
  if(state.otpProcessing) return;
  const code = collectOtpValue_();
  if(code.length !== OTP_LENGTH){
    state.otpError = 'Please enter the complete ' + OTP_LENGTH + '-digit verification code.';
    render();
    return;
  }
  state.otpProcessing = true; state.otpError = null; render();
  google.script.run.withSuccessHandler(function(res){
    state.otpProcessing = false;
    if(res.success){
      stopOtpCountdown_();
      state.user = res.user;
      state.otpEmail = ''; state.otpMasked = ''; state.otpExpiresAt = 0;
      showToast('Email verified successfully.', 'success');
      enterDashboard();
    } else {
      markOtpBoxesError_(true);
      state.otpError = res.message || 'Invalid OTP. Please enter the correct verification code.';
      render();
      if(res.expired || res.locked){
        // Code can no longer be used — clear the boxes and let them request a fresh one.
        for(let i=0;i<OTP_LENGTH;i++){ const el=document.getElementById('otpBox'+i); if(el) el.value=''; }
        state.otpExpiresAt = 0;
        updateOtpCountdownUI_();
      } else {
        focusOtpBox_(0);
      }
    }
  }).withFailureHandler(function(err){
    state.otpProcessing = false;
    console.error('verifyEmailOtp failed:', err);
    state.otpError = 'Connection error. Please try again.';
    markOtpBoxesError_(true);
    render();
  }).verifyEmailOtp(state.user.userId, code, detectDevice(), clientIp);
}

function onResendOtp(){
  if(!state.otpResendReady || state.otpProcessing) return;
  state.otpProcessing = true; state.otpError = null; render();
  google.script.run.withSuccessHandler(function(res){
    state.otpProcessing = false;
    if(res.success){
      state.otpMasked = res.maskedEmail || state.otpMasked;
      state.otpExpiresAt = res.expiresAt;
      for(let i=0;i<OTP_LENGTH;i++){ const el=document.getElementById('otpBox'+i); if(el){ el.value=''; el.classList.remove('error'); } }
      render();
      startOtpCountdown_();
      setTimeout(function(){ focusOtpBox_(0); }, 60);
      showToast('A new verification code has been sent.', 'success');
    } else {
      state.otpError = res.message;
      if(res.expiresAt){ state.otpExpiresAt = res.expiresAt; }
      render();
    }
  }).withFailureHandler(function(){
    state.otpProcessing = false;
    state.otpError = 'Connection error. Please try again.';
    render();
  }).resendEmailOtp(state.user.userId, detectDevice(), clientIp);
}

let otpCountdownHandle = null;
function startOtpCountdown_(){
  stopOtpCountdown_();
  state.otpResendReady = false;
  updateOtpCountdownUI_();
  otpCountdownHandle = setInterval(updateOtpCountdownUI_, 1000);
}
function stopOtpCountdown_(){
  clearInterval(otpCountdownHandle); otpCountdownHandle = null;
}
function updateOtpCountdownUI_(){
  const remaining = state.otpExpiresAt - Date.now();
  const label = document.getElementById('otpCountdownLabel');
  const resendBtn = document.getElementById('otpResendBtn');
  if(remaining <= 0){
    state.otpResendReady = true;
    stopOtpCountdown_();
    if(label) label.textContent = '00:00';
    if(resendBtn) resendBtn.disabled = false;
  } else {
    state.otpResendReady = false;
    if(label) label.textContent = fmtCountdown(remaining);
    if(resendBtn) resendBtn.disabled = true;
  }
}

let sigCtx=null, sigDrawing=false, sigHasContent=false;
function initSigPad(){
  const canvas = document.getElementById('sigPad'); if(!canvas) return;
  const ratio = window.devicePixelRatio||1;
  canvas.width = canvas.clientWidth*ratio; canvas.height = canvas.clientHeight*ratio;
  sigCtx = canvas.getContext('2d'); sigCtx.scale(ratio,ratio);
  sigCtx.lineWidth=2.2; sigCtx.lineCap='round'; sigCtx.strokeStyle='#0d2b4e';
  function pos(e){ const r=canvas.getBoundingClientRect(); const cx=e.touches?e.touches[0].clientX:e.clientX; const cy=e.touches?e.touches[0].clientY:e.clientY; return {x:cx-r.left,y:cy-r.top}; }
  function start(e){ sigDrawing=true; const p=pos(e); sigCtx.beginPath(); sigCtx.moveTo(p.x,p.y); e.preventDefault(); }
  function move(e){ if(!sigDrawing) return; const p=pos(e); sigCtx.lineTo(p.x,p.y); sigCtx.stroke(); sigHasContent=true; updateSubmitState(); e.preventDefault(); }
  function end(){ sigDrawing=false; }
  canvas.onmousedown=start; canvas.onmousemove=move; window.addEventListener('mouseup',end);
  canvas.ontouchstart=start; canvas.ontouchmove=move; canvas.ontouchend=end;
}
function clearSig(){ const canvas=document.getElementById('sigPad'); if(sigCtx&&canvas) sigCtx.clearRect(0,0,canvas.width,canvas.height); sigHasContent=false; updateSubmitState(); }

function withSaveTimeout_(key, onTimeout){
  const timer = setTimeout(function(){
    if(state.settingsSaving[key]){
      state.settingsSaving[key] = false;
      render();
      showToast('This is taking longer than expected — please try again.', 'error');
      onTimeout && onTimeout();
    }
  }, SETTING_SAVE_TIMEOUT_MS);
  return timer;
}

/* ============================================================
   BACKGROUND AUTO-SYNC (silent, every ~10-15s)
   ============================================================ */
let syncTimer = null;
function startAutoSync(){
  stopAutoSync();
  syncTimer = setInterval(silentSync, SYNC_INTERVAL_MS);
}

function silentSync(){
  if(!state.user || !state.token) return;
  if(state.processing || state.lockInfo || state.rejectModal || state.voteConfirm) return;
  if(state.view === 'adminSettings' || state.view === 'register') return;
  if(state.view === 'login' || state.view === 'boot' || state.view === 'email_gate' || state.view === 'otp_verify' || state.view === 'dashboardLoading') return;

  google.script.run.withSuccessHandler(function(snap){
    if(!snap) return;
    let changed = false;
    function upd(key){
      if(!(key in snap)) return;
      const nv = JSON.stringify(snap[key]);
      const ov = JSON.stringify(state[key]);
      if(nv !== ov){ state[key] = snap[key]; changed = true; }
    }
    upd('settings');
    upd('notifications');
    if('notifHistory' in snap){
      const nv = JSON.stringify(snap.notifHistory);
      const ov = JSON.stringify(state.notifHistory);
      if(nv !== ov){ state.notifHistory = snap.notifHistory; recomputeUnreadCount_(); changed = true; }
    }
    if(!state.statusLoading && !state.registerLoading && state.view !== 'register' && 'myNomination' in snap){
      const incoming = snap.myNomination;
      const current = state.myNomination;
      const nv = JSON.stringify(incoming);
      const ov = JSON.stringify(current);
      if(nv !== ov && incoming){
        applyMyNomination(incoming);
        changed = true;
        if(state.view === 'status'){
          state.statusUpdating = true;
          setTimeout(function(){ state.statusUpdating = false; softRender(); }, 600);
        }
      }
    }
    if(!state.voteLoading){ upd('ballot'); upd('myVote'); }
    if(!state.resultsLoading) upd('results');
    if(!state.approverLoading) upd('approverInbox');
    if('adminNominations' in snap && !(snap.adminNominations && snap.adminNominations.error)) upd('adminNominations');
    if('adminStats' in snap && !(snap.adminStats && snap.adminStats.error)) upd('adminStats');
    if(state.view === 'dashboard' && (isAdmin() || state.activeRole === 'Viewer') && !('adminStats' in snap)) loadAdminStats();

    if(changed) softRender();
  }).withFailureHandler(function(){ /* silent — never disrupt the UI */ })
    .getSyncSnapshot(state.user.userId, state.view, state.approverLevel);
}

function softRender(){
  const y = window.scrollY;
  render();
  window.scrollTo(0, y);
}

/* ============================================================
   I4 — Employee Dashboard live countdown for scheduled tiles.
   When Registration/Voting is off purely because of a future
   "Opens at" schedule (not a manual Admin off-switch), the tile
   shows a clock icon + the scheduled date/time + a ticking
   countdown (rendered directly via DOM text updates, not a full
   re-render, mirroring the existing OTP/lock countdown pattern).
   Once any visible countdown reaches zero, a silent sync is
   triggered so the tile unlocks automatically without a manual
   page refresh.
   ============================================================ */
let dashboardCountdownHandle = null;
function startDashboardCountdown_(){
  stopDashboardCountdown_();
  updateDashboardCountdowns_();
  dashboardCountdownHandle = setInterval(updateDashboardCountdowns_, 1000);
}
function stopDashboardCountdown_(){
  clearInterval(dashboardCountdownHandle); dashboardCountdownHandle = null;
}
function fmtCountdownLong_(ms){
  const s = Math.floor(ms/1000);
  const d = Math.floor(s/86400);
  const h = Math.floor((s%86400)/3600);
  const m = Math.floor((s%3600)/60);
  const sec = s%60;
  if(d>0) return d+'d '+String(h).padStart(2,'0')+'h '+String(m).padStart(2,'0')+'m';
  return String(h).padStart(2,'0')+':'+String(m).padStart(2,'0')+':'+String(sec).padStart(2,'0');
}
function updateDashboardCountdowns_(){
  const els = document.querySelectorAll('.tile-countdown[data-open]');
  if(!els.length) return;
  let anyReached = false;
  els.forEach(function(el){
    const openAt = el.getAttribute('data-open');
    if(!openAt) return;
    const target = new Date(openAt).getTime();
    if(isNaN(target)) return;
    const remaining = target - Date.now();
    if(remaining <= 0){
      el.textContent = 'Opening now…';
      anyReached = true;
    } else {
      el.textContent = 'Opens in ' + fmtCountdownLong_(remaining);
    }
  });
  if(anyReached){
    stopDashboardCountdown_();
    setTimeout(function(){ silentSync(); }, 1200);
  }
}

function tileMode_(tv, group, key){
  const v = tv && tv[group] ? tv[group][key] : undefined;
  if(v === false || v === 'hidden') return 'hidden';
  if(v === 'disabled') return 'disabled';
  return 'visible';
}

/* I4 — determines whether a tile is closed purely because of a
   future "Opens at" schedule (manual switch is still ON), in which
   case we show a live countdown instead of a flat "disabled" label. */
function scheduleInfo_(manualEnabled, windowMessage, openAt){
  if(!manualEnabled) return null; // manually turned off by Admin — no countdown to show
  if(openAt && windowMessage && windowMessage.indexOf('Opens') === 0){
    return { openAt: openAt, label: windowMessage };
  }
  return null;
}

function StepTracker(status){
  const steps=[{k:'submitted',l:'Submitted'},{k:'a1',l:'Approver 1'},{k:'a2',l:'Approver 2'},{k:'ballot',l:'On Ballot'}];
  let idx=0;
  if(status==='Pending Approver 1') idx=1;
  else if(status==='Pending Approver 2') idx=2;
  else if(status==='Approved') idx=3;
  else if(status==='Rejected') idx=1;
  return `<div class="steps">${steps.map((s,i)=>`<div class="step ${i<idx?'done':''} ${i===idx?'active':''}"><div class="dot">${i<idx?ic('check',13):i+1}</div><span>${s.l}</span></div>`).join('')}</div>`;
}

function LockModal(){

  const remaining = state.lockInfo.until - Date.now();
  return `<div class="modal-backdrop"><div class="modal">
    <h3>${ic('lock',18)} Account Temporarily Locked</h3>

/* ---------- root render ---------- */
function render(){
  let html='';
  switch(state.view){
    case 'boot': html=BootView(); break;
    case 'login': html=LoginView(); break;
    case 'email_gate': html=EmailGateView(); break;
    case 'otp_verify': html=OtpVerifyView(); break;
    case 'dashboardLoading': html=DashboardLoadingView(); break;
    case 'dashboard': html=DashboardView(); break;
    case 'register': html = state.registerLoading ? RegisterSkeletonView() : RegisterView(); break;
    case 'status': html = state.statusLoading ? StatusSkeletonView() : StatusView(); break;
    case 'vote': html = state.voteLoading ? VoteSkeletonView() : VoteView(); break;
    case 'results': html = state.resultsLoading ? ResultsSkeletonView() : ResultsView(); break;
    case 'approver': html = state.approverLoading ? ApproverSkeletonView() : ApproverInboxView(); break;
    case 'adminManage': html=AdminManageView(); break;
    case 'adminSettings': html=AdminSettingsView(); break;
    case 'adminReports': html=AdminReportsView(); break;
    default: html=DashboardView();
  }
  html += ProcessingOverlay() + LockModal() + RejectModalEl() + VoteConfirmModalEl() + ToastEl();
  if(state.view==='register' && !state.registerLoading && !state.myNomination && !state.eligibilityAcknowledged) html += EligibilityOverlay();
  document.getElementById('app').innerHTML = html;
  if(state.view==='register' && !state.registerLoading && !state.myNomination) updateIntroCounter();
  if(state.view==='otp_verify'){ updateOtpCountdownUI_(); }
  if(state.view==='dashboard'){ startDashboardCountdown_(); } else { stopDashboardCountdown_(); }
}

/* Clicking anywhere outside the notification panel closes it. */

  if(state.notifOpen && !e.target.closest('.notif-wrap')){
    state.notifOpen = false; render();
  }
});
